#version 330 core

in vec2  UV;
in float diffuse;
in vec3  ambient;
in bool  lightType;

out vec3 ready_pixel;
uniform sampler2D TextureSampler;

void main(){
  if (lightType == 0)
     ready_pixel = texture( TextureSampler, UV ).rgb * ambient;
  else
     ready_pixel = texture( TextureSampler, UV ).rgb * ambient * diffuse;
}
